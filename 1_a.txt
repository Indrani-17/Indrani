#include <iostream>
using namespace std;
int main()
{
    string s;
    cin>>s;
    int a=0,i=0;
    while(s[i]!='\0'){
        char c=s[i];
        if(s[i]!='a'&&s[i]!='b'){
            cout<<"Rejected";
            return 0;
        } 
        switch(a){
            case 0: if(c=='a')
                    a=1;
                    else if(c=='b')
                    a=3;
                    else
                    cout<<"Invalid character...\n";
                    i++;
                    break;
            case 1: if(c=='a')
                    a=0;
                    else if(c=='b')
                    a=2;
                    else
                    cout<<"Invalid character...\n";
                    i++;
                    break;
            case 2: if(c=='a')
                    a=3;
                    else if(c=='b')
                    a=1;
                    else
                    cout<<"Invalid character...\n";
                    i++;
                    break;
            case 3: if(c=='a')
                    a=2;
                    else if(c=='b')
                    a=0;
                    else
                    cout<<"Invalid character...\n";
                    i++;
                    break;
        }
    }
    if(a==0)
    cout<<"Accepted\n";
    else
    cout<<"Rejected\n";
    return 0;
}
